pub fn bucket_sort<T, F>(arr: Vec<T>, mapper: F) -> Result<Vec<T>, &'static str>
    where F: Fn(&T) -> f64
{
    let arr_length = arr.len();

    buckets.resize_with(arr_length, || Vec::new());

    // è¿›æ¡¶

        let key = mapper(&e);
        if key >= 0.0 && key < 1.0 {
            let bucket_id = (key * (arr_length as f64)).approx_as::<usize>().unwrap();

    // æ’åºæ¯ä¸ªæ¡¶ï¼Œå¹¶è¿æ¥

        .collect();
    Ok(result)
}
