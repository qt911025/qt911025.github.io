    pub fn sort<Elem: PartialOrd>(vec: &mut Vec<Elem>, ascend: bool) {
    for i in 1..vec.len() {
        unsafe {
            let e = ptr::read(&vec[i]);
            let mut j = i;
            if ascend {
                while j > 0 && &vec[j - 1] > &e {
                    ptr::copy(&vec[j - 1], &mut vec[j], 1);
                    j = j - 1;
                }
            } else {
                while j > 0 && &vec[j - 1] < &e {
                    ptr::copy(&vec[j - 1], &mut vec[j], 1);
                    j = j - 1;
                }
            }
            vec[j] = e;
        }
    }
}
